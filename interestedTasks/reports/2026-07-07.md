### Блок 5: Предскажи вывод

```javascript
function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn(...args);
    }
    return (...next) => curried(...args, ...next);
  };
}

const sum = (a, b, c) => a + b + c;

curry(sum)(1)(2, 3);       // ?
curry(sum)(1, 2, 3, 99);   // ?
```
